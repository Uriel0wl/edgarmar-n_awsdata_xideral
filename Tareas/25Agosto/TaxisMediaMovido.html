import io      # Para manejar flujos de datos en memoria (buffer)
import pandas as pd  # Para manipulacin y anlisis de datos

# Crear un cliente de S3 para interactuar con Amazon S3
s3 = boto3.client("s3")

# Definir la funcin principal de Lambda
def lambda_handler(event, context):
    # Crear un buffer en memoria para descargar el archivo
    buffer = io.BytesIO()
    
    # Descargar el archivo Parquet desde S3 al buffer en memoria
    s3.download_fileobj(
        Bucket="xideralaws-curso-uriel",  # Nombre del bucket S3
        Key="nyc_taxi_2023/yellow_tripdata_2023-01.parquet",  # Ruta del archivo
        Fileobj=buffer  # Objeto donde se guardarn los datos
    )

    # Mover el puntero al inicio del buffer para poder leerlo
    buffer.seek(0)
    
    # Cargar el archivo Parquet en un DataFrame de pandas
    df = pd.read_parquet(buffer, engine="pyarrow")

    # Limpiar los datos:
    # Eliminar filas con valores faltantes (NaN)
    df = df.dropna()
    # Eliminar filas duplicadas
    df = df.drop_duplicates()

    # Calcular promedios solo para columnas numricas
    # Identificar columnas con tipos numricos
    numeric_columns = df.select_dtypes(include=['number']).columns
    means_dict = {}  # Diccionario para almacenar los promedios

    # Calcular el promedio para cada columna numrica
    for column in numeric_columns:
        means_dict[column] = float(df[column].mean())

    # Devolver la respuesta HTTP con los resultados
    return {
        "statusCode": 200,  # Cdigo de estado HTTP (xito)
        "body": json.dumps(means_dict, default=str)  # Convertir resultados a JSON

# Importar las libreras necesarias
import json       # Para trabajar con datos en formato JSON
import boto3      # SDK de AWS para interactuar con servicios de AWS
import io         # Para manejar flujos de datos en memoria (buffer)
import pandas as pd  # Para manipulacin y anlisis de datos

# Crear un cliente de S3 para interactuar con Amazon S3
s3 = boto3.client("s3")

# Definir la funcin principal de Lambda
def lambda_handler(event, context):
    # Crear un buffer en memoria para descargar el archivo
    buffer = io.BytesIO()
    
    # Descargar el archivo Parquet desde S3 al buffer en memoria
    s3.download_fileobj(
        Bucket="xideralaws-curso-uriel",  # Nombre del bucket S3
        Key="nyc_taxi_2023/yellow_tripdata_2023-01.parquet",  # Ruta del archivo
        Fileobj=buffer  # Objeto donde se guardarn los datos
    )

    # Mover el puntero al inicio del buffer para poder leerlo
    buffer.seek(0)
    
    # Cargar el archivo Parquet en un DataFrame de pandas
    df = pd.read_parquet(buffer, engine="pyarrow")

    # Limpiar los datos:
    # Eliminar filas con valores faltantes (NaN)
    df = df.dropna()
    # Eliminar filas duplicadas
    df = df.drop_duplicates()

    # Calcular promedios solo para columnas numricas
    # Identificar columnas con tipos numricos
    numeric_columns = df.select_dtypes(include=['number']).columns
    means_dict = {}  # Diccionario para almacenar los promedios

    # Calcular el promedio para cada columna numrica
    for column in numeric_columns:
        means_dict[column] = float(df[column].mean())

    # Crear el JSON con los resultados del anlisis
    result = {
        "message": "Promedios de columnas calculados exitosamente",
        "total_registros": len(df),
        "promedios_columnas": means_dict
    }

    # Guardar el JSON resultante en S3 con una nueva ruta
    s3.put_object(
        Bucket="xideralaws-curso-uriel",  # Bucket de destino
        Key="nyc_taxi_2023_json_promedio/taxi_means_analysis.json",  # Nueva ruta y nombre del archivo
        Body=json.dumps(result, indent=2, default=str),  # JSON formateado
        ContentType='application/json'  # Tipo de contenido
    )

    # Devolver respuesta de xito
    return {
        "statusCode": 200,  # Cdigo de estado HTTP (xito)
        "body": json.dumps({
            "message": "Anlisis completado y guardado en S3",
            "ubicacion_s3": "s3://xideralaws-curso-uriel/nyc_taxi_2023_json_promedio/taxi_means_analysis.json"
        }, default=str)
    }
